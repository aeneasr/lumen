Ollama `all-minilm`, GitHub Actions services, `//go:build e2e` tag.

package main

import (

)

var serverBinary string

func TestMain(m *testing.M) {
	// Build the server binary.
	bin := filepath.Join(os.TempDir(), fmt.Sprintf("agent-index-e2e-%d", os.Getpid()))
	cmd := exec.Command("go", "build", "-o", bin, ".")
	cmd.Env = append(os.Environ(), "CGO_ENABLED=1")
	if out, err := cmd.CombinedOutput(); err != nil {
		fmt.Fprintf(os.Stderr, "build failed: %v\n%s\n", err, out)
		os.Exit(1)
	}
	serverBinary = bin
	defer os.Remove(bin)

	// Check Ollama health.
	ollamaHost := envOrDefault("OLLAMA_HOST", "http://localhost:11434")
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()
	req, _ := http.NewRequestWithContext(ctx, "GET", ollamaHost, nil)
	if _, err := http.DefaultClient.Do(req); err != nil {
		fmt.Fprintf(os.Stderr, "ollama not reachable at %s: %v\n", ollamaHost, err)
		os.Exit(1)
	}

	os.Exit(m.Run())
}

// startServer launches the MCP server binary as a subprocess and returns a
// connected MCP client session. The server gets its own temp DB directory.
func startServer(t *testing.T) *mcp.ClientSession {
	t.Helper()

	ollamaHost := envOrDefault("OLLAMA_HOST", "http://localhost:11434")
	model := envOrDefault("AGENT_INDEX_EMBED_MODEL", "all-minilm")
	dims := envOrDefault("AGENT_INDEX_EMBED_DIMS", "384")
	dataDir := t.TempDir()

	cmd := exec.Command(serverBinary)
	cmd.Env = []string{
		"OLLAMA_HOST=" + ollamaHost,
		"AGENT_INDEX_EMBED_MODEL=" + model,
		"AGENT_INDEX_EMBED_DIMS=" + dims,
		"XDG_DATA_HOME=" + dataDir,
		"HOME=" + os.Getenv("HOME"),
		"PATH=" + os.Getenv("PATH"),
	}

	client := mcp.NewClient(&mcp.Implementation{
		Name:    "e2e-test",
		Version: "0.0.1",
	}, nil)

	ctx := context.Background()
	session, err := client.Connect(ctx, &mcp.CommandTransport{Command: cmd}, nil)
	if err != nil {
		t.Fatalf("connect to server: %v", err)
	}
	t.Cleanup(func() { session.Close() })

	return session
}

// callSearch calls the semantic_search tool and returns parsed output.
func callSearch(t *testing.T, session *mcp.ClientSession, args map[string]any) SemanticSearchOutput {
	t.Helper()
	result, err := session.CallTool(context.Background(), &mcp.CallToolParams{
		Name:      "semantic_search",
		Arguments: args,
	})
	if err != nil {
		t.Fatalf("CallTool semantic_search: %v", err)
	}
	if result.IsError {
		t.Fatalf("semantic_search returned error: %v", result.Content)
	}
	raw, ok := result.StructuredContent.(json.RawMessage)
	if !ok {
		t.Fatalf("expected json.RawMessage, got %T", result.StructuredContent)
	}
	var out SemanticSearchOutput
	if err := json.Unmarshal(raw, &out); err != nil {
		t.Fatalf("unmarshal search output: %v", err)
	}
	return out
}

func TestE2E_IndexAndSearch(t *testing.T) {
	session := startServer(t)
	projectPath := sampleProjectPath(t)

	out := callSearch(t, session, map[string]any{
		"query": "authentication token validation",
		"path":  projectPath,
		"limit": 5,
	})

	if !out.Reindexed {
		t.Error("expected reindexed=true on first search")
	}
	if len(out.Results) == 0 {
		t.Fatal("expected at least 1 result")
	}

	// The top results should include auth-related symbols.
	foundAuth := false
	for _, r := range out.Results[:min(3, len(out.Results))] {
		t.Logf("result: %s %s:%d-%d (score=%.4f)", r.Kind, r.FilePath, r.StartLine, r.EndLine, r.Score)
		if r.Symbol == "ValidateToken" || r.Symbol == "CreateSession" || r.Symbol == "RevokeSession" {
			foundAuth = true
		}
	}
	if !foundAuth {
		t.Error("expected an auth function in top-3 results for 'authentication token validation'")
	}
}

func TestE2E_IncrementalUpdate(t *testing.T) {
	session := startServer(t)

	// Copy fixture to a temp dir so we can modify it.
	projectDir := t.TempDir()
	srcDir := sampleProjectPath(t)
	entries, err := os.ReadDir(srcDir)
	if err != nil {
		t.Fatal(err)
	}
	for _, e := range entries {
		data, err := os.ReadFile(filepath.Join(srcDir, e.Name()))
		if err != nil {
			t.Fatal(err)
		}
		if err := os.WriteFile(filepath.Join(projectDir, e.Name()), data, 0o644); err != nil {
			t.Fatal(err)
		}
	}

	// Initial index.
	out1 := callSearch(t, session, map[string]any{
		"query": "graceful shutdown",
		"path":  projectDir,
		"limit": 5,
	})
	if !out1.Reindexed {
		t.Error("expected initial reindex")
	}

	// Add a new file with a GracefulShutdown function.
	newFile := filepath.Join(projectDir, "shutdown.go")
	if err := os.WriteFile(newFile, []byte(`package project

// GracefulShutdown stops the server gracefully, draining active connections.
func GracefulShutdown(timeout int) error {
	return nil
}
`), 0o644); err != nil {
		t.Fatal(err)
	}

	// Search again — should pick up the new file.
	out2 := callSearch(t, session, map[string]any{
		"query": "graceful shutdown",
		"path":  projectDir,
		"limit": 5,
	})
	if !out2.Reindexed {
		t.Error("expected reindex after adding file")
	}

	found := false
	for _, r := range out2.Results {
		if r.Symbol == "GracefulShutdown" {
			found = true
			break
		}
	}
	if !found {
		t.Error("expected GracefulShutdown in results after incremental update")
	}
}
